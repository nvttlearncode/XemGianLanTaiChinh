// DataVec ETL Pipeline — chuẩn hóa features từ CSV Kaggle Schema schema = new Schema.Builder() .addColumnDouble("Time") .addColumnsDouble("V1", "V2", "V3", ..., "V28") .addColumnDouble("Amount") .addColumnInteger("Class") // 0=normal, 1=fraud (chỉ dùng khi eval) .build(); TransformProcess tp = new TransformProcess.Builder(schema) .removeColumns("Time", "Class") .normalize(new NormalizerMinMaxScaler(0, 1)) .build(); // Chạy ETL trên Spark cluster (284k giao dịch) JavaRDD<List<Writable>> rawData = sc.textFile("hdfs:///data/creditcard.csv"); JavaRDD<List<Writable>> processed = SparkTransformExecutor.execute(rawData, tp); // Chỉ lấy giao dịch bình thường để train Autoencoder JavaRDD<DataSet> normalData = processed .filter(row -> row.get(28).toInt() == 0) .map(row -> new DataSet(toINDArray(row), toINDArray(row)));
// Spring Boot REST endpoint — inference real-time @RestController @RequestMapping("/api/fraud") public class FraudDetectionController { @Autowired private MultiLayerNetwork autoencoderModel; @PostMapping("/predict") public FraudResult predict(@RequestBody TransactionDTO tx) { // 1. Vector hóa bằng ND4J INDArray input = Nd4j.create(tx.toFeatureArray()); // 2. Reconstruction qua Autoencoder INDArray reconstructed = autoencoderModel.output(input); // 3. Tính reconstruction error (MSE) double mse = input.squaredDistance(reconstructed) / input.length(); // 4. Threshold: MSE > 0.025 → nghi ngờ gian lận boolean isFraud = mse > 0.025; double score = Math.min(mse / 0.05 * 100, 100); return new FraudResult(isFraud, score, mse); } }
